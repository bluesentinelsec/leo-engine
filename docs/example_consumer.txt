cmake_minimum_required(VERSION 3.25)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

project(myapp
    VERSION 0.1.0
    DESCRIPTION "A minimal executable"
    LANGUAGES C CXX
)

set(CMAKE_C_STANDARD 17)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(LEO_BUILD_SHARED ON CACHE BOOL "Build Leo as a shared library")
set(LEO_BUILD_TESTS ON CACHE BOOL "Build unit tests")

include(FetchContent)
FetchContent_Declare(
    leo
    GIT_REPOSITORY https://github.com/bluesentinelsec/leo-engine.git
    GIT_TAG develop
)
FetchContent_MakeAvailable(leo)

add_executable(myapp main.cpp)
target_link_libraries(myapp PRIVATE Leo::Runtime)

# Copy shared library to the executable's directory
if(LEO_BUILD_SHARED)
    add_custom_command(
        TARGET myapp POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
                $<TARGET_FILE:Leo::Runtime>
                $<TARGET_FILE_DIR:myapp>
        COMMENT "Copying leo.dll to executable directory"
    )
endif()
